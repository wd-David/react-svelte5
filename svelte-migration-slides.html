Pattern 1: Svelte in React

useSvelte Hook

// 1. Install dependencies pnpm add -D svelte @sveltejs/vite-plugin-svelte // 2. Create wrapper hook export default function useSvelte(Component: Component) { const svelteRef = useRef(null); return (props: Props) => { useLayoutEffect(() => { mount(Component, { target: svelteRef.current, props, }); }, []); return
; }; } // 3. Use with memo for performance export default memo(function MyComponent(props) { const SvelteComponent = useSvelte(Component); return ; });

Key: Use memo() to prevent unnecessary remounts

Pattern 2: React in SvelteKit

use-react.svelte Component

<script lang="ts"> import { onMount } from "svelte"; import { createRoot } from "react-dom/client"; import React from "react"; let { component, ...reactProps } = $props(); let container = $state<HTMLDivElement>(); let root = $state<ReturnType<typeof createRoot>>(); onMount(() => { if (container) { root = createRoot(container); root.render(React.createElement(component, reactProps)); } return () => root?.unmount(); }); </script> <div bind:this={container}></div>

Modern State Pattern

Classes + Context + Runes

// my-state.svelte.ts import { FiniteStateMachine, watch, resource } from 'runed' export class MyState extends FiniteStateMachine { readonly myResource: ResourceReturn constructor(getter: () => string) { super("off", { off: { toggle: "on" }, on: { toggle: "off" } }) // Use watch instead of $effect for better control watch(getter, () => {}) // Or use resource for async operations this.myResource = resource(getter, async () => {}) } } // Context API for sharing const KEY = Symbol('my-state'); export const setMyState = (g) => setContext(KEY, new MyState(g)); export const getMyState = () => getContext<MyState>(KEY);

Pattern inspired by @huntabyte's "Classes + Context"
